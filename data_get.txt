import requests
from bs4 import BeautifulSoup

BASE_URL = "https://test-directory.srl.info/akiruno/test/list/{}"

all_results = []

for i in range(1, 73):
    url = BASE_URL.format(i)
    r = requests.get(url)
    r.raise_for_status()

    soup = BeautifulSoup(r.text, "html.parser")
    table = soup.select_one(
        "div.list-container-div:not(.clone) table.list_culomn9"
    )

    if not table:
        print(f"no table: {url}")
        continue

    tbody = table.find("tbody")
    rows = tbody.find_all("tr", class_="with_link")

    for tr in rows:
        item_name = tr.find("th").get_text(strip=True)
        detail_url = tr.get("link_url")

        all_results.append({
            "category_id": i,
            "item_name": item_name,
            "detail_url": detail_url
        })

print(len(all_results))

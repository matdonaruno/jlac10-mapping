function main(workbook: ExcelScript.Workbook) {

    const srcSheet = workbook.getActiveWorksheet();
    const srcName: string = srcSheet.getName();
    const dstName: string = `${srcName}_Barcode`;

    // 既存シート削除
    const oldSheet = workbook.getWorksheet(dstName);
    if (oldSheet) {
        oldSheet.delete();
    }

    const dstSheet = workbook.addWorksheet(dstName);

    // ===== 設定 =====
    const START_ROW = 3;
    const ROW_HEIGHT = 120;
    const BARCODE_FONT = "Code 128";

    // 元シート列
    const COL_REQUEST = 1; // B
    const COL_ITEMNAME = 2; // C
    const COL_JLAC10 = 4; // E

    const srcRange = srcSheet.getUsedRange();
    if (!srcRange) return;

    const rowCount: number = srcRange.getRowCount();

    // ===== ヘッダー =====
    dstSheet.getRange("A1").setValue("設定依頼");
    dstSheet.getRange("B1").setValue("設定依頼（Barcode）");
    dstSheet.getRange("C1").setValue("項目名称");
    dstSheet.getRange("D1").setValue("JLAC10");
    dstSheet.getRange("E1").setValue("JLAC10（Barcode）");

    let outRow = 1;

    for (let r = START_ROW; r < rowCount; r++) {

        const reqValue = srcSheet.getCell(r, COL_REQUEST).getValue();
        const itemValue = srcSheet.getCell(r, COL_ITEMNAME).getValue();
        const jlacValue = srcSheet.getCell(r, COL_JLAC10).getValue();

        // どちらも空ならスキップ
        if (reqValue === "" && jlacValue === "") continue;

        const reqNorm = normalizeToAlphaNumeric(reqValue);
        const jlacNorm = normalizeToAlphaNumeric(jlacValue);

        // 通常文字
        dstSheet.getCell(outRow, 0).setValue(reqNorm);
        dstSheet.getCell(outRow, 2).setValue(
            typeof itemValue === "string" ? itemValue : ""
        );
        dstSheet.getCell(outRow, 3).setValue(jlacNorm);

        // バーコード
        if (reqNorm !== "") {
            const cell = dstSheet.getCell(outRow, 1);
            cell.setValue(toCode128(reqNorm));
            applyBarcodeStyle(cell, BARCODE_FONT);
        }

        if (jlacNorm !== "") {
            const cell = dstSheet.getCell(outRow, 4);
            cell.setValue(toCode128(jlacNorm));
            applyBarcodeStyle(cell, BARCODE_FONT);
        }

        // 行高さ固定
        dstSheet.getRangeByIndexes(outRow, 0, 1, 5)
            .getFormat()
            .setRowHeight(ROW_HEIGHT);

        outRow++;
    }

    // 列幅
    dstSheet.getRange("A:A").getFormat().setColumnWidth(120);
    dstSheet.getRange("B:B").getFormat().setColumnWidth(280);
    dstSheet.getRange("C:C").getFormat().setColumnWidth(200);
    dstSheet.getRange("D:D").getFormat().setColumnWidth(260);
    dstSheet.getRange("E:E").getFormat().setColumnWidth(300);

    // ページ設定
    const pageLayout = dstSheet.getPageLayout();
    pageLayout.setOrientation(ExcelScript.PageOrientation.portrait);
    pageLayout.setZoom({ scale: 100 });
    pageLayout.setLeftMargin(10);
    pageLayout.setRightMargin(10);
    pageLayout.setTopMargin(10);
    pageLayout.setBottomMargin(10);
    pageLayout.setHeaderMargin(5);
    pageLayout.setFooterMargin(5);


    dstSheet.activate();
}


/**
 * Excel値を安全に英数字へ正規化
 */
function normalizeToAlphaNumeric(
    value: string | number | boolean
): string {
    if (typeof value !== "string") return "";

    return value
        .normalize("NFKC")            // 全角→半角
        .replace(/[^0-9A-Za-z]/g, "") // 英数字以外除外
        .trim();
}

/**
 * Code128 ラッパー
 */
function toCode128(value: string): string {
    return `Ì${value}Î`;
}

/**
 * バーコード書式
 */
function applyBarcodeStyle(
    cell: ExcelScript.Range,
    fontName: string
): void {
    const f = cell.getFormat();
    f.getFont().setName(fontName);
    f.getFont().setSize(24);
    f.setHorizontalAlignment(ExcelScript.HorizontalAlignment.center);
    f.setVerticalAlignment(ExcelScript.VerticalAlignment.center);
}
